SELECT I.SUMMARY AS ISSUE_SUMMARY, IL.NAME AS ISSUELIST_NAME FROM issues I 
JOIN issueslists IL ON IL.ID = I.ISSUESLIST_ID;

2. 
SELECT I.SUMMARY AS ISSUE_SUMMARY, U.FIRSTNAME, U.LASTNAME FROM issues I
JOIN users U ON U.ID = I.USER_ID_ASSIGNEDTO;

3.
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_NUMBER from issues I
JOIN users U ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
